393  SUBROUTINE spectral_x_derivative(xLenSub, yLenSub, specXDeriv, order)
394 
395  IMPLICIT NONE
396 
397  INTEGER(qb), INTENT(IN) :: xlensub
398  INTEGER(qb), INTENT(IN) :: ylensub
399  INTEGER(qb), INTENT(IN) :: order
400  COMPLEX(dp), INTENT(INOUT) :: specxderiv(xlensub,ylensub)
401  INTEGER(qb) :: i
402  REAL(dp) :: xwavenums(xlensub)
404 
405  ! Calculate the x-direction wavenumbers.
406  xwavenums = 0.0_dp
407  xwavenums(2) = 1.0_dp
408  xwavenums(xlensub) = -1.0_dp
409  DO i = 1_qb, xlensub/2_qb - 1_qb
410  xwavenums(xlensub - i) = real(-i - 1_qb, dp)
411  xwavenums(i + 2_qb) = real(i + 1_qb, dp)
412  END DO
413  ! If xLenSub even and order odd, we have to zero out highest wavenumber for
414  ! derivative.
415  IF ((mod(xlensub, 2_qb) .EQ. 0_qb) .AND. (mod(order, 2_qb) .EQ. 1_qb)) THEN
416  xwavenums(xlensub/2_qb + 1_qb) = 0.0_dp
417  END IF
418 
419  DO i = 1, ylensub
420  specxderiv(:,i) = (cmplx(0.0, xwavenums(:), dp))**(real(order, dp))
421  END DO
422 
423  END SUBROUTINE spectral_x_derivative
