55  ! Stability constraint derived from our time step method, which is Forward
56  ! Euler in time, and a 1st order centered difference for each spatial
57  ! derivative (using Von Neumann analysis).
58  cflcondition = 1.0_dp/(2.0_dp * (xdiffus/(dx**2.0_dp) &
59  + ydiffus/(dy**2.0_dp)))
60 
61  IF ((dt .GT. cflcondition)) THEN
62  print *, "WARNING: Time step size ", dt, " exceeds the CFL parameter ", &
63  cflcondition, ". Simulation may be unstable."
64  END IF
65 
66  stepparity = 0_qb
67  step = 0_qb
68  time = inittime
69  maxstep = int((fintime - inittime)/dt,qb)
70 
71  IF (maxstep .LT. outputfreq) THEN
72  error stop 'Output frequency too large. Please reduce.'
73  END IF
74 
75  ! Output the initial condition.
76  IF (outputfreq .NE. 0) THEN
77  CALL write_output(step, stepparity)
78  END IF
79 
80  ! Step simulation forward in time.
81  DO WHILE (time .LE. fintime)
82  stepparity = 2_qb - (stepparity + 1_qb)
83  CALL time_step_scheme(stepparity)
84  time = time + dt
85  step = step + 1_qb
86 
87  ! Write output at desired frequency.

90  CALL write_output(step, stepparity)
91  END IF

93  END DO
